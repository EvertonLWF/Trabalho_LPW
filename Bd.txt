create table posts(
	id_user int not null,
	titulo varchar(255),
	descricao varchar(255),
	data date,
	id_post int,
	primary key(id_post),
	foreign key(id_user) references usuario (id_user)

);

create table usuario(
	id_user int not null,
	email varchar(255),
	senha varchar(255),
	nome varchar(255),
	tipo int,
	primary key(id_user)


);

create table categoria (
	nm_categoria varchar(255) not null,
	primary key (id_categoria)
);
