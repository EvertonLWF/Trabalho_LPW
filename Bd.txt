create database Blog;

create table posts(
	id_user int not null,
	titulo varchar(255),
	descricao varchar(255),
	data date,
	id_post int,
	categoria varchar(255),
	primary key(id_post),
	constraind fk_post foreingh key(id_user) references id_user(usuario)

);

create table usuario(
	id_user int not null,
	email varchar(255),
	senha varchar(255),
	nome varchar(255),
	primary key(id_user)


);
